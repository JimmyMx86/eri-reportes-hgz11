Paso 2 — Crear el Apps Script (Web App)

En tu hoja de cálculo, abre Extensiones → Apps Script.
Borra todo el contenido del editor y pega exactamente este código:
const SHEET_NAME = 'Hoja1'; function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const p = e.parameter; sheet.appendRow([ p.folio, p.fecha, p.hora, p.tipo, p.paciente, p.nss, p.cama, p.umf, p.edad, p.medico, p.residente, p.ta, p.fc, p.fr, p.temp, p.sato2, p.diagnosticos, p.exploracion, p.labs, p.plan, p.estado, p.reporte ]); const result = JSON.stringify({ok: true}); const cb = p.callback; if (cb) { return ContentService.createTextOutput(cb + '(' + result + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(result) .setMimeType(ContentService.MimeType.JSON); } catch(err) { const result = JSON.stringify({ok: false, error: err.toString()}); const cb = e.parameter.callback; if (cb) { return ContentService.createTextOutput(cb + '(' + result + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(result) .setMimeType(ContentService.MimeType.JSON); } } } catch(err) { }
